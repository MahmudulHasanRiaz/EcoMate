1) Prisma schema (new models)
File: prisma/schema.prisma
Add:
```
model MarketingCampaign {
  id          String   @id @default(cuid())
  name        String
  businessId  String?
  marketerId  String?
  channel     String?
  objective   String?
  status      String? // Draft/Active/Paused/Completed
  budget      Float   @default(0)
  startDate   DateTime? @db.Date
  endDate     DateTime? @db.Date
  notes       String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  business  Business?    @relation(fields: [businessId], references: [id], onDelete: SetNull)
  marketer  StaffMember? @relation(fields: [marketerId], references: [id], onDelete: SetNull)
  spends    MarketingSpend[]
  attributions MarketingAttribution[]

  @@index([businessId, status])
  @@index([marketerId])
  @@index([startDate, endDate])
}

model MarketingSpend {
  id          String   @id @default(cuid())
  campaignId  String
  businessId  String?
  date        DateTime @db.Date
  amount      Float
  notes       String?
  createdById String?
  expenseId   String? // future link
  createdAt   DateTime @default(now())

  campaign MarketingCampaign @relation(fields: [campaignId], references: [id], onDelete: Cascade)
  business Business? @relation(fields: [businessId], references: [id], onDelete: SetNull)
  createdBy StaffMember? @relation(fields: [createdById], references: [id], onDelete: SetNull)

  @@index([campaignId, date])
  @@index([businessId, date])
}

model MarketingAttribution {
  id          String   @id @default(cuid())
  campaignId  String
  orderId     String
  businessId  String?
  marketerId  String?
  attributedAt DateTime @default(now())

  campaign MarketingCampaign @relation(fields: [campaignId], references: [id], onDelete: Cascade)
  order    Order @relation(fields: [orderId], references: [id], onDelete: Cascade)
  business Business? @relation(fields: [businessId], references: [id], onDelete: SetNull)
  marketer StaffMember? @relation(fields: [marketerId], references: [id], onDelete: SetNull)

  @@unique([campaignId, orderId])
  @@index([businessId])
  @@index([marketerId])
}
```
Run migration + generate.
