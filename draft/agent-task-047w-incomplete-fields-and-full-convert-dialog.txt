P47w - Incomplete Leads: Full Field Capture + Full Convert Dialog

Goal
- Incomplete leads must capture whatever the user fills (name, email, address, note), not only phone.
- Convert must use the SAME full New Order dialog (product picker, delivery ratio, etc.).

2) New API: resolve SKUs into actual order items
--------------------------------------------------------------------

Create file: src/app/api/orders/incomplete/resolve-skus/route.ts

Purpose
- Convert lead SKUs into concrete order items for NewOrderDialog prefill.

Implementation

```
import prisma from '@/lib/prisma';
import { apiError, apiServerError, apiSuccess } from '@/lib/error';
import { enforcePermission } from '@/lib/security';

export const runtime = 'nodejs';
export const dynamic = 'force-dynamic';

function normalizeSkuList(input: any): string[] {
  if (!input) return [];
  if (Array.isArray(input)) {
    return input.map((x) => (typeof x === 'string' ? x : x?.sku || x?.SKU || '')).filter(Boolean);
  }
  if (typeof input === 'string') {
    return input.split(',').map((s) => s.trim()).filter(Boolean);
  }
  return [];
}

export async function POST(req: Request) {
  try {
    const { allowed, error } = await enforcePermission('orders', 'read');
    if (!allowed) return error;

    const body = await req.json().catch(() => ({}));
    const skuList = normalizeSkuList(body?.skuList);
    if (!skuList.length) return apiSuccess({ items: [], missing: [] });

    const variants = await prisma.productVariant.findMany({
      where: { sku: { in: skuList } },
      select: { id: true, sku: true, productId: true, price: true, salePrice: true, name: true, image: true }
    });

    const variantSkuSet = new Set(variants.map(v => v.sku));
    const productSkus = skuList.filter(sku => !variantSkuSet.has(sku));

    const products = productSkus.length
      ? await prisma.product.findMany({
          where: { sku: { in: productSkus } },
          select: { id: true, sku: true, name: true, price: true, salePrice: true, image: true }
        })
      : [];

    const productBySku = new Map(products.map(p => [p.sku, p]));

    const items = skuList.flatMap((sku) => {
      const variant = variants.find(v => v.sku === sku);
      if (variant) {
        const price = Number(variant.salePrice ?? variant.price ?? 0);
        return [{
          productId: variant.productId,
          variantId: variant.id,
          sku: variant.sku,
          name: variant.name || variant.sku,
          price,
          quantity: 1,
          image: variant.image || ''
        }];
      }
      const product = productBySku.get(sku);
      if (product) {
        const price = Number(product.salePrice ?? product.price ?? 0);
        return [{
          productId: product.id,
          variantId: null,
          sku: product.sku,
          name: product.name,
          price,
          quantity: 1,
          image: product.image || ''
        }];
      }
      return [];
    });

    const foundSkuSet = new Set(items.map(i => i.sku));
    const missing = skuList.filter(sku => !foundSkuSet.has(sku));

    return apiSuccess({ items, missing });
  } catch (e: any) {
    console.error('[API:INCOMPLETE_RESOLVE_SKUS]', e);
    return apiServerError(e);
  }
}
```

A. Add functions near incomplete helpers:

```
export async function resolveIncompleteSkus(skuList: string[]) {
  const cookieHeader = await getServerCookieHeader();
  const res = await fetch(`${API_BASE_URL}/orders/incomplete/resolve-skus`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      ...(cookieHeader ? { Cookie: cookieHeader } : {}),
    },
    credentials: 'include',
    body: JSON.stringify({ skuList }),
    next: { revalidate: 0 },
  });
  const json = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(json?.message || json?.error || 'Failed to resolve SKUs');
  return json?.data || json;
}

export async function markIncompleteLeadConverted(leadId: string, orderId: string) {
  const cookieHeader = await getServerCookieHeader();
  const res = await fetch(`${API_BASE_URL}/orders/incomplete/${leadId}`, {
    method: 'PATCH',
    headers: {
      'Content-Type': 'application/json',
      ...(cookieHeader ? { Cookie: cookieHeader } : {}),
    },
    credentials: 'include',
    body: JSON.stringify({ action: 'converted', orderId }),
  });
  const json = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(json?.message || json?.error || 'Failed to mark lead converted');
  return json?.data || json;
}
```
