```
const CONFIG_CACHE_KEY = 'fashionary_config_cache_v1';
const CONFIG_TTL_MS = 10 * 60 * 1000;

function loadCachedConfig() {
  try {
    const raw = localStorage.getItem(CONFIG_CACHE_KEY);
    if (!raw) return null;
    const parsed = JSON.parse(raw);
    if (!parsed || !parsed.data || !parsed.savedAt) return null;
    if (Date.now() - parsed.savedAt > CONFIG_TTL_MS) return null;
    return parsed.data;
  } catch { return null; }
}

function saveCachedConfig(cfg) {
  try { localStorage.setItem(CONFIG_CACHE_KEY, JSON.stringify({ data: cfg, savedAt: Date.now() })); } catch {}
}
```

```
function openRestrictionModal(message, supportPhone) {
  ensureRestrictionModal();
  const modal = document.getElementById('fashionary-restriction-modal');
  const text = document.getElementById('fashionary-restriction-text');
  const call = document.getElementById('fashionary-restriction-call');
  text.textContent = supportPhone ? `${message} (Support: ${supportPhone})` : message;
  if (supportPhone) {
    call.style.display = 'inline-flex';
    call.href = 'tel:' + supportPhone;
  } else {
    call.style.display = 'none';
  }
  modal.style.display = 'flex';
}

function closeRestrictionModal() {
  const modal = document.getElementById('fashionary-restriction-modal');
  if (modal) modal.style.display = 'none';
}

function renderRestrictionInline($input, message, supportPhone) {
  const $field = getFieldWrapper($input);
  const text = supportPhone ? `${message} (Support: ${supportPhone})` : message;
  let $msg = $field.find('.fashionary-restriction');
  if (!$msg.length) {
    $msg = $('<div class="fashionary-restriction" style="color:#b91c1c;font-size:12px;margin-top:6px;"></div>');
    $field.append($msg);
  }
  $msg.text(text);
}

function clearRestrictionInline($input) {
  const $field = getFieldWrapper($input);
  $field.find('.fashionary-restriction').remove();
}
```

3) Restriction-check API call + balanced enforcement
Add below helpers (before blockIfInvalid):

```
async function checkRestriction(phone) {
  if (!config || !config.restrictionEnabled) return { allowed: true };
  if (!API_URL || !API_KEY) return { allowed: true };

  const normalized = String(phone || '').trim();
  if (!normalized) return { allowed: true };
  if (normalized === restrictionLastPhone) return { allowed: !restrictionBlocked };
  restrictionLastPhone = normalized;

  try {
    const res = await fetch(API_URL + '/restriction-check', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer ' + API_KEY
      },
      body: JSON.stringify({ phone: normalized })
    });
    const data = await res.json().catch(() => ({}));
    if (!res.ok) return { allowed: true };
    return data;
  } catch (e) {
    console.warn('[FASHIONARY_RESTRICTION_ERR]', e);
    // balanced mode: allow if API fails
    return { allowed: true, error: 'api_fail' };
  }
}
```

```
function blockIfInvalid(e) {
  const phone = findPhoneInput().val() || '';
  if (!isPhoneValid(phone)) {
    const msg = (config && config.phoneValidation && config.phoneValidation.message) || 'Please enter a valid phone number before placing the order.';
    openRestrictionModal(msg, '');
    if (e && e.preventDefault) e.preventDefault();
    if (e && e.stopImmediatePropagation) e.stopImmediatePropagation();
    return false;
  }
  if (restrictionBlocked) {
    const msg = restrictionMessage || (config?.restrictionMessage || 'You are restricted from placing orders.');
    openRestrictionModal(msg, restrictionSupportPhone || config?.supportPhone || '');
    if (e && e.preventDefault) e.preventDefault();
    if (e && e.stopImmediatePropagation) e.stopImmediatePropagation();
    return false;
  }
}
```
