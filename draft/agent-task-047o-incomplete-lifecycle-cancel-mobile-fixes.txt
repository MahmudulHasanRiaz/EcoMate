```ts
const RECENT_COMPLETION_TTL_SEC = 30 * 60;

function completionKey(integrationId: string, phone: string) {
  return `woo:lead:completed:${integrationId}:${phone}`;
}

async function markRecentCompletion(integrationId: string, phone: string) {
  try {
    const redis = getRedisClient();
    if (redis) {
      await redis.set(completionKey(integrationId, phone), '1', 'EX', RECENT_COMPLETION_TTL_SEC);
    }
  } catch (e) {
    console.error('[INCOMPLETE_CONVERT_RECENT_COMPLETION_ERR]', e);
  }
}
```

3) Cancel must persist (add PATCH on incomplete lead)
File: src/app/api/orders/incomplete/[id]/route.ts

Add this PATCH handler below GET:

```ts
export async function PATCH(req: Request, { params }: { params: { id: string } }) {
  try {
    const { allowed, error } = await enforcePermission('orders', 'delete');
    if (!allowed) return error;

    const body = await req.json().catch(() => ({}));
    const action = String(body?.action || 'cancel').toLowerCase();
    if (action !== 'cancel') return apiError('Unsupported action', 400);

    const lead = await prisma.wooCheckoutLead.update({
      where: { id: params.id },
      data: {
        status: 'CANCELLED',
        completedAt: new Date(),
      },
    });

    return apiSuccess({ id: lead.id, status: lead.status });
  } catch (e: any) {
    console.error('[API:INCOMPLETE_LEAD_PATCH]', e);
    return apiServerError(e);
  }
}
```

Notes:
- This route already imports apiError/apiSuccess/apiServerError + prisma + enforcePermission.
- CANCELLED leads will naturally disappear because the list filters status=OPEN.

4) UI: make Cancel persistent + add View/Convert on mobile cards

A) Replace handleCancel with a persistent version:

```ts
const handleCancel = async (leadId: string) => {
  try {
    const res = await fetch(`/api/orders/incomplete/${leadId}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ action: 'cancel' }),
    });
    const json = await res.json().catch(() => ({}));
    if (!res.ok) throw new Error(json?.message || json?.error || 'Cancel failed');

    setAllOrders((prev) => prev.filter((o) => o.id !== leadId));
    toast.success('Lead cancelled');
  } catch (e: any) {
    toast.error(e?.message || 'Failed to cancel lead');
  }
};
```
