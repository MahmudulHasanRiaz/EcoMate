UPDATE P20 - Short-TTL cache for report/stats endpoints (Redis + memory fallback)

Scope: ONLY add caching for report/stats endpoints. Keep changes minimal. No refactors.

Goal:
- Cache heavy report/stat responses for 120s (2 min) to reduce load.
- Use Redis if configured; fallback to in-memory Map.
- Cache key must include staffId and query params so access scope is safe.

Add helper
File: src/server/utils/report-cache.ts (new)
```
import { getRedisClient } from '@/server/queues/redis';

const CACHE_TTL_MS = 120 * 1000;
const CACHE_TTL_SEC = Math.floor(CACHE_TTL_MS / 1000);
const mem = new Map<string, { expires: number; data: unknown }>();

export async function getReportCache<T>(key: string, loader: () => Promise<T>, ttlMs: number = CACHE_TTL_MS): Promise<T> {
  const now = Date.now();
  const cached = mem.get(key);
  if (cached && cached.expires > now) return cached.data as T;

  const redis = getRedisClient();
  if (redis) {
    try {
      const raw = await redis.get(key);
      if (raw) {
        const parsed = JSON.parse(raw) as T;
        mem.set(key, { expires: now + ttlMs, data: parsed });
        return parsed;
      }
    } catch (err) {
      console.warn('[REPORT_CACHE_REDIS_READ_FAIL]', err);
    }
  }

  const data = await loader();
  mem.set(key, { expires: now + ttlMs, data });
  if (redis) {
    try {
      await redis.set(key, JSON.stringify(data), 'EX', Math.floor(ttlMs / 1000));
    } catch (err) {
      console.warn('[REPORT_CACHE_REDIS_WRITE_FAIL]', err);
    }
  }
  return data;
}
```
