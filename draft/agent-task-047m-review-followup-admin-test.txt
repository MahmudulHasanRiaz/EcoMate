1) Implement the admin test handler
File: `draft/fashionary-woo-plugin/assets/js/admin.js`
Add the following block UNDER the existing ōTest Connectionö handler:

```js
    $('#fashionary-test-incomplete').on('click', function () {
        const apiUrl = $('input[name="fashionary_api_url"]').val();
        const apiKey = $('input[name="fashionary_api_key"]').val();
        const $status = $('#fashionary-connection-status');

        $status.text('Testing incomplete capture...').removeClass('success error');

        if (!apiUrl || !apiKey) {
            $status.text('Missing API URL or Key').addClass('error');
            return;
        }

        const url = apiUrl + '/incomplete-orders?apiKey=' + encodeURIComponent(apiKey);
        const payload = {
            debug: true,
            phone: '01700000000',
            name: 'Admin Debug',
            address: 'Debug Address',
            skuList: ['DEBUG-SKU']
        };

        fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'text/plain;charset=UTF-8' },
            body: JSON.stringify(payload),
            mode: 'cors',
            credentials: 'omit'
        })
            .then(async (res) => {
                if (!res.ok) {
                    const t = await res.text().catch(() => '');
                    throw new Error(t || `HTTP ${res.status}`);
                }
                return res.json().catch(() => ({}));
            })
            .then(() => {
                $status.text('Incomplete test OK (debug)').addClass('success');
            })
            .catch((err) => {
                console.error('[FASHIONARY_ADMIN_INCOMPLETE_TEST_ERR]', err);
                $status.text('Incomplete test failed').addClass('error');
            });
    });
```
