3) Plugin admin JS: wire the “Test Incomplete” button
File: draft/fashionary-woo-plugin/assets/js/admin.js

Add this handler below the connection test:
```
$('#fashionary-test-incomplete').on('click', function () {
    const apiUrl = $('input[name="fashionary_api_url"]').val();
    const apiKey = $('input[name="fashionary_api_key"]').val();
    const $status = $('#fashionary-connection-status');

    $status.text('Testing incomplete...').removeClass('success error');

    if (!apiUrl || !apiKey) {
        $status.text('Missing API URL or Key').addClass('error');
        return;
    }

    const url = apiUrl + '/incomplete-orders?apiKey=' + encodeURIComponent(apiKey);
    const payload = {
        debug: true,
        phone: '01700000000',
        name: 'Debug Test',
        address: 'Debug Address',
        skuList: []
    };

    $.ajax({
        url,
        method: 'POST',
        contentType: 'text/plain; charset=UTF-8',
        data: JSON.stringify(payload),
        success: function () {
            $status.text('Incomplete endpoint OK').addClass('success');
        },
        error: function (xhr) {
            $status.text('Incomplete test failed: ' + (xhr.status || 'ERR')).addClass('error');
        }
    });
});
```

Expected result:
- Clicking Test Incomplete should show success.
- Server log should show: [WOO_INCOMPLETE_DEBUG_OK] <integrationId>

B) Improve findPhoneInput to prefer billing_phone but fall back to any phone-ish field:
Replace findPhoneInput() with:
```
function findPhoneInput() {
    // Prefer canonical billing_phone
    let $el = $('#billing_phone').first();
    if ($el.length) return $el;

    $el = $('input[name="billing_phone"]').first();
    if ($el.length) return $el;

    // Fallback: any phone-like input
    for (const sel of PHONE_SELECTORS) {
        const $cand = $(sel).filter(':visible').first();
        if ($cand.length) return $cand;
    }
    return $();
}
```

D) Add logging on attempted capture reasons:
Replace attemptCapture() with:
```
function attemptCapture(useBeacon) {
    if (!config || !config.incompleteEnabled) {
        console.log('[FASHIONARY_CAPTURE_SKIP]', 'config-disabled');
        return;
    }
    const $input = findPhoneInput();
    if (!$input.length) {
        console.log('[FASHIONARY_CAPTURE_SKIP]', 'no-phone-input');
        return;
    }
    const phone = String($input.val() || '');
    if (!isPhoneValid(phone)) {
        console.log('[FASHIONARY_CAPTURE_SKIP]', 'invalid-phone', phone);
        return;
    }
    console.log('[FASHIONARY_CAPTURE_ATTEMPT]', useBeacon ? 'beacon' : 'debounce');
    captureIncomplete(phone, useBeacon);
}
```
